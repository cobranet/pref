def self.create(players)
  p = Prefgame.new
  #create hand
  h = Hand.new
  h.create
  #create game
  @full_game = Game.new(h.get_cards)
  p.game = @full_game.to_game_string
  p.east = players[0]
  p.south = players[1]
  p.west = players[2]
  p.save!
  p
end
