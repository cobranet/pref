  if Waiting.find_by_user_id(user_id) == nil

def self.set_up_game
  prefgame = nil
  waitings = Waiting.find(:all)
  counter = 0
  for_game = Array.new
  waitings.each do |w| 
    counter = counter + 1
    for_game << w.user_id
      if counter == 3 
        prefgame = Prefgame.create(for_game)
        for_game.each do |user_id|
          w = Waiting.find_by_user_id(user_id)
          w.destroy   
      end     
    end
  end
  prefgame
end
